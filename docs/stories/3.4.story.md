## Notas Técnicas

```tsx
interface FullscreenActionsProps {
  video: VideoItem;
  settings: WidgetConfig['settings'];
  isLiked: boolean;
  onToggleLike: () => void;
}

// WhatsApp URL:
const waUrl = generateWhatsAppUrl(
  video.whatsapp || settings.whatsappDefault,
  video.product.name,
  video.product.price,
  video.product.url
);

// Share:
const handleShare = async () => {
  if (navigator.share) {
    await navigator.share({ title: video.product.name, url: video.product.url });
  } else {
    await navigator.clipboard.writeText(video.product.url);
    // mostrar feedback: "Link copiado!"
  }
};
```
