# Story 6.2 — API Routes Next.js: CRUD /api/videos + /api/settings

**Epic:** 6 — Painel de Administração
**Status:** Draft
**Prioridade:** Must
**Estimativa:** 2h

---

## User Story

Como desenvolvedor,
Quero API routes Next.js para CRUD completo de vídeos e configurações,
Para que o admin e o widget consumam dados de forma centralizada.

---

## Acceptance Criteria

- [ ] **AC-1:** `GET /api/videos` — retorna array de vídeos do KV
- [ ] **AC-2:** `POST /api/videos` — adiciona novo vídeo (requer auth)
- [ ] **AC-3:** `PUT /api/videos/[id]` — atualiza vídeo existente (requer auth)
- [ ] **AC-4:** `DELETE /api/videos/[id]` — remove vídeo (requer auth)
- [ ] **AC-5:** `PUT /api/videos/reorder` — reordena array de vídeos (requer auth)
- [ ] **AC-6:** `GET /api/settings` — retorna configurações gerais
- [ ] **AC-7:** `PUT /api/settings` — atualiza configurações (requer auth)
- [ ] **AC-8:** Rotas de escrita verificam cookie `admin_token` via `jwtVerify`
- [ ] **AC-9:** `GET /api/videos` e `GET /api/settings` são públicas (usadas pelo widget)
- [ ] **AC-10:** Validação de campos obrigatórios com resposta 400 em caso de erro
- [ ] **AC-11:** Fallback: se KV não disponível, lê de `videos.json` (graceful degradation)

## Notas Técnicas

```typescript
// src/lib/auth.ts — verificar auth em rotas protegidas
import { jwtVerify } from 'jose';
import { cookies } from 'next/headers';

export async function requireAuth(): Promise<boolean> {
  const token = cookies().get('admin_token')?.value;
  if (!token) return false;
  try {
    const secret = new TextEncoder().encode(process.env.JWT_SECRET!);
    await jwtVerify(token, secret);
    return true;
  } catch {
    return false;
  }
}
```

```typescript
// src/lib/kv.ts — abstração KV com fallback JSON
import { kv } from '@vercel/kv';
import videosJson from '@/data/videos.json';

const VIDEOS_KEY = 'vcw:videos';
const SETTINGS_KEY = 'vcw:settings';

export async function getVideos(): Promise<VideoItem[]> {
  try {
    const videos = await kv.get<VideoItem[]>(VIDEOS_KEY);
    return videos ?? videosJson.videos;
  } catch {
    return videosJson.videos; // fallback para JSON estático
  }
}

export async function getSettings(): Promise<WidgetConfig['settings']> {
  try {
    const settings = await kv.get<WidgetConfig['settings']>(SETTINGS_KEY);
    return settings ?? videosJson.settings;
  } catch {
    return videosJson.settings;
  }
}

export async function setVideos(videos: VideoItem[]): Promise<void> {
  await kv.set(VIDEOS_KEY, videos);
}

export async function setSettings(settings: WidgetConfig['settings']): Promise<void> {
  await kv.set(SETTINGS_KEY, settings);
}
```

```typescript
// /api/videos/route.ts
export async function GET() {
  const videos = await getVideos();
  return Response.json(videos);
}

export async function POST(request: Request) {
  if (!(await requireAuth())) return Response.json({ error: 'Unauthorized' }, { status: 401 });
  const video = await request.json() as VideoItem;
  // Validar campos obrigatórios
  if (!video.id || !video.videoUrl || !video.product?.name) {
    return Response.json({ error: 'Campos obrigatórios faltando' }, { status: 400 });
  }
  const videos = await getVideos();
  videos.push(video);
  await setVideos(videos);
  return Response.json(video, { status: 201 });
}
```
