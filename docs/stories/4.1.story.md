## User Story

Como desenvolvedor que instala o widget,
Quero um script que monta o React app automaticamente no DOM,
Para instalar com apenas 2 linhas de HTML.

## Notas Técnicas

```typescript
// mount.ts
import React from 'react';
import ReactDOM from 'react-dom/client';
import { VideoShowcase } from './embed';

(async () => {
  // 1. Find or create mount point
  let root = document.getElementById('video-commerce-widget');
  if (!root) {
    root = document.createElement('div');
    root.id = 'video-commerce-widget';
    document.body.appendChild(root);
  }

  // 2. Inject Swiper CSS
  if (!document.querySelector('link[data-swiper-css]')) {
    const link = document.createElement('link');
    link.rel = 'stylesheet';
    link.href = 'https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.css';
    link.setAttribute('data-swiper-css', 'true');
    document.head.appendChild(link);
  }

  // 3. Load config
  let config;
  try {
    // Option A: inline config
    if (window.__VIDEO_WIDGET_CONFIG__) {
      config = window.__VIDEO_WIDGET_CONFIG__;
    } else {
      // Option B: fetch from same origin
      const res = await fetch('/api/widget-config');
      config = await res.json();
    }
  } catch (e) {
    console.error('[VideoWidget] Failed to load config:', e);
    return;
  }

  // 4. Mount
  ReactDOM.createRoot(root).render(
    React.createElement(VideoShowcase, { config })
  );
})();
```
