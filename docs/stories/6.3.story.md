## Notas Técnicas

```typescript
// scripts/seed-kv.ts
import { kv } from '@vercel/kv';
import { config } from 'dotenv';
import videosData from '../src/data/videos.json';

config({ path: '.env.local' });

async function seed() {
  console.log('Seeding Vercel KV...');

  const existingVideos = await kv.get('vcw:videos');
  if (!existingVideos) {
    await kv.set('vcw:videos', videosData.videos);
    console.log(`✅ Seeded ${videosData.videos.length} videos`);
  } else {
    console.log('⏭️  Videos already exist in KV, skipping');
  }

  const existingSettings = await kv.get('vcw:settings');
  if (!existingSettings) {
    await kv.set('vcw:settings', videosData.settings);
    console.log('✅ Seeded settings');
  } else {
    console.log('⏭️  Settings already exist in KV, skipping');
  }

  console.log('Done!');
  process.exit(0);
}

seed().catch(console.error);
```

### Setup Vercel KV (para README)
```bash
# 1. No Vercel Dashboard:
#    Project → Storage → Create Database → KV

# 2. Conectar ao projeto:
#    KV Database → Settings → Connect Project

# 3. Puxar variáveis de ambiente:
vercel env pull .env.local

# 4. Fazer seed inicial:
npm run seed
```

### Variáveis de Ambiente Necessárias
```
KV_REST_API_URL=https://xxx.kv.vercel-storage.com
KV_REST_API_TOKEN=xxx
KV_REST_API_READ_ONLY_TOKEN=xxx  # opcional
```
